ECharts封装核心代码

/**
 * 渲染ECharts图表（高级封装版）
 * @param {string} containerId - 图表容器ID
 * @param {string} chartType - 图表类型 (bar, line, pie, radar, etc.)
 * @param {object} data - 图表数据
 * @param {object} [options] - 额外配置选项
 * @returns {echarts.ECharts} - ECharts实例
 */
function renderAdvancedChart(containerId, chartType, data, options = {}) {
    // 初始化图表实例
    const chart = echarts.init(document.getElementById(containerId));
    
    // 基础配置（深色主题优化）
    const baseConfig = {
        backgroundColor: 'transparent',
        tooltip: {
            trigger: 'item',
            backgroundColor: 'rgba(30, 30, 48, 0.9)',
            borderColor: 'rgba(0, 168, 255, 0.5)',
            textStyle: { color: '#fff' }
        },
        toolbox: {
            feature: {
                saveAsImage: { title: '保存为图片' },
                dataView: { title: '数据视图' }
            }
        },
        grid: {
            left: '3%',
            right: '4%',
            bottom: '12%',
            containLabel: true
        },
        textStyle: {
            color: 'rgba(255, 255, 255, 0.7)'
        }
    };
    
    // 图表类型配置
    let typeConfig = {};
    
    switch (chartType) {
        case 'bar':
            typeConfig = {
                xAxis: {
                    type: 'category',
                    data: data.categories,
                    axisLine: { lineStyle: { color: 'rgba(255, 255, 255, 0.2)' } }
                },
                yAxis: {
                    type: 'value',
                    axisLine: { lineStyle: { color: 'rgba(255, 255, 255, 0.2)' } }
                },
                series: [{
                    type: 'bar',
                    data: data.values,
                    itemStyle: {
                        color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
                            { offset: 0, color: '#00a8ff' },
                            { offset: 1, color: '#0062ff' }
                        ])
                    }
                }]
            };
            break;
            
        // 其他图表类型配置省略...
    }
    
    // 合并配置
    const finalConfig = {
        ...baseConfig,
        ...typeConfig,
        ...options,
        title: {
            text: options.title || '',
            left: 'center',
            textStyle: {
                color: '#fff',
                fontSize: 20
            }
        }
    };
    
    // 渲染图表
    chart.setOption(finalConfig);
    
    // 响应式调整
    const resizeHandler = () => chart.resize();
    window.addEventListener('resize', resizeHandler);
    
    return {
        chart,
        updateData: (newData) => {
            // 更新数据的逻辑...
        }
    };
}
