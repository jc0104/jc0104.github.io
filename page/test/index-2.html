实现原理

以下是我们封装的ECharts组件核心代码，通过简单参数配置即可生成图表：

/**
 * 渲染ECharts图表
 * @param {string} containerId - 图表容器元素ID
 * @param {string} chartType - 图表类型 (bar, line, pie, radar, etc.)
 * @param {object} data - 图表数据
 * @param {object} options - 额外配置选项
 */
function renderChart(containerId, chartType, data, options = {}) {
    // 初始化图表实例
    const chart = echarts.init(document.getElementById(containerId));
    
    // 基础配置
    const baseConfig = {
        tooltip: {
            trigger: 'item'
        },
        toolbox: {
            feature: {
                saveAsImage: {}
            }
        },
        grid: {
            left: '3%',
            right: '4%',
            bottom: '3%',
            containLabel: true
        }
    };
    
    // 根据图表类型添加特定配置
    let typeConfig = {};
    
    switch (chartType) {
        case 'bar':
            typeConfig = {
                xAxis: {
                    type: 'category',
                    data: data.categories
                },
                yAxis: {
                    type: 'value'
                },
                series: [{
                    type: 'bar',
                    data: data.values
                }]
            };
            break;
            
        case 'line':
            typeConfig = {
                xAxis: {
                    type: 'category',
                    data: data.categories
                },
                yAxis: {
                    type: 'value'
                },
                series: [{
                    type: 'line',
                    data: data.values,
                    smooth: true
                }]
            };
            break;
            
        case 'pie':
            typeConfig = {
                series: [{
                    type: 'pie',
                    radius: '50%',
                    data: data.items,
                    emphasis: {
                        itemStyle: {
                            shadowBlur: 10,
                            shadowOffsetX: 0,
                            shadowColor: 'rgba(0, 0, 0, 0.5)'
                        }
                    }
                }]
            };
            break;
            
        case 'radar':
            typeConfig = {
                radar: {
                    indicator: data.indicators
                },
                series: [{
                    type: 'radar',
                    data: data.values
                }]
            };
            break;
    }
    
    // 合并配置
    const finalConfig = {
        ...baseConfig,
        ...typeConfig,
        ...options,
        title: {
            text: options.title || '',
            left: 'center'
        },
        legend: {
            orient: 'horizontal',
            bottom: 'bottom'
        }
    };
    
    // 应用配置并渲染图表
    chart.setOption(finalConfig);
    
    // 响应窗口大小变化
    window.addEventListener('resize', () => {
        chart.resize();
    });
    
    return chart;
}
